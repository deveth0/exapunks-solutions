; READ 16 NUMBERS FROM M
MARK LOOP
MAKE
@REP 16
COPY M F
@END

; TRIGGER NEXT EXA
REPL LOOP

; ITERATE THROUGH VALUES
SEEK -9999
@REP 8
; ODD
MULI F 2 T
ADDI T X X
; SUB 9 IF OVERFLOW
TEST T > 9
FJMP EVEN@{0,1}
SUBI X 9 X

; EVEN
MARK EVEN@{0,1}
ADDI F X X
@END
; MODI IS 0 FOR HIT
MODI X 10 T
FJMP FOUND
WIPE
HALT

MARK FOUND
; CLEANUP OTHER EXAS
DROP
LINK 800
LINK 802
LINK 799
KILL
LINK -1
LINK -1
LINK -1
KILL
